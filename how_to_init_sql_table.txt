# how to init sql tables for dailymed

CREATE TABLE DAILYMED_USER (
    USER    INT AUTO_INCREMENT PRIMARY KEY,
    MAKE    DATETIME,
    LAST    DATETIME
);

CREATE TABLE DAILYMED_MEDI (
    MEDI    INT AUTO_INCREMENT PRIMARY KEY,
    USER    INT NOT NULL,
    NAME    CHAR(10),
    DATE    DATETIME,
    TERM    INT,
    WNTK    INT,
    ICON    INT,
    COLR    INT
);

CREATE TABLE DAILYMED_DATA (
    AKEY     INT AUTO_INCREMENT PRIMARY KEY,
    MEDI    INT,
    USER    INT NOT NULL,
    TAKE    DATETIME
);
